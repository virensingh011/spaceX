  <script>
    // ===== WEATHER =====
    async function getWeather() {
      const city = document.getElementById("city").value;
      if (!city) return;

      const apiKey = "3318d63e265b4b46895105610262103"; // your key
      const url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&units=metric&appid=${apiKey}`;

      try {
        const res = await fetch(url);
        const data = await res.json();
        if (data.cod !== 200) {
          document.getElementById("weatherResult").innerText = "City not found!";
          return;
        }

        document.getElementById("weatherResult").innerHTML = `
          <div class="card">
            <h3>${data.name} Weather</h3>
            <p>☀️ ${data.weather[0].description}</p>
            <p>🌡 Temperature: ${data.main.temp}°C</p>
            <p>💧 Humidity: ${data.main.humidity}%</p>
          </div>
        `;
      } catch (e) {
        console.error(e);
      }
    }

    // ===== NEWS =====
    async function loadNews() {
      const newsList = document.getElementById("newsList");
      newsList.innerHTML = "Loading...";

      const url = "https://api.spaceflightnewsapi.net/v3/articles?_limit=6";

      try {
        const res = await fetch(url);
        const articles = await res.json();

        newsList.innerHTML = "";
        articles.forEach(item => {
          newsList.innerHTML += `
            <div class="news-item card">
              <h4><a href="${item.url}" target="_blank" style="color:#81ecec;">${item.title}</a></h4>
              <p>${item.summary.substring(0, 140)}...</p>
            </div>
          `;
        });
      } catch (e) {
        newsList.innerText = "Failed to load news.";
      }
    }
    loadNews();

    // ===== AI ASSISTANT =====
    async function askAI() {
      const prompt = document.getElementById("aiQuestion").value;
      const out = document.getElementById("aiAnswer");
      if (!prompt) return;

      out.innerText = "Thinking... 🚀";

      const apiKey = "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"; // your Hugging Face key

      try {
        const res = await fetch("https://api-inference.huggingface.co/models/gpt2", {
          method: "POST",
          headers: {
            Authorization: `Bearer ${apiKey}`,
            "Content-Type": "application/json"
          },
          body: JSON.stringify({ inputs: prompt })
        });
        const data = await res.json();
        out.innerText = data[0]?.generated_text || "No response!";
      } catch (e) {
        out.innerText = "AI Error!";
        console.error(e);
      }
    }
  </script>
